def Connect():
    """
    Connects to the MySQL database.

    Returns:
        mysql.connector.connection.MySQLConnection: The database connection object.
    """
    conn = None
    try:
        conn = mysql.connector.Connect(
            host='localhost',
            user='root',
            passwd='root',
            database='futelibs'
        )
    except:
        print('Error', sys.exc_info())
    finally:
        return conn
